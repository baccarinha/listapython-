Maior e Menor Valor

"""Maior e menor valor"""

valores = []

for i in range(10):
    num = float(input(f"Digite o {i+1}º valor: "))
    valores.append(num)

maior = max(valores)
menor = min(valores)

print("Maior valor:", maior)
print("Menor valor:", menor)

Lista Ordenada (sem sort)

"""Desafio"""

valores = []

for i in range(10):
    num = float(input(f"Digite o {i+1}º valor: "))

    if len(valores) == 0:
        valores.append(num)
    else:
        inserido = False

        for j in range(len(valores)):
            if num < valores[j]:
                valores.insert(j, num)
                inserido = True
                break

        if not inserido:
            valores.append(num)

print("Lista ordenada:", valores)
