Calculadora Simples

"""Calculadora simples"""

num1 = float(input("Digite o primeiro número: "))
num2 = float(input("Digite o segundo número: "))

soma = num1 + num2
subtracao = num1 - num2
multiplicacao = num1 * num2

if num2 != 0:
    divisao = num1 / num2
else:
    divisao = "Não é possível dividir por zero"

print("Soma:", soma)
print("Subtração:", subtracao)
print("Multiplicação:", multiplicacao)
print("Divisão:", divisao)

Área de Figuras

"""Área de figuras"""

base = float(input("Digite a base do retângulo: "))
altura = float(input("Digite a altura do retângulo: "))
area_retangulo = base * altura

raio = float(input("Digite o raio do círculo: "))
pi = 3.14
area_circulo = pi * (raio ** 2)

print("Área do retângulo:", area_retangulo)
print("Área do círculo:", area_circulo)

Soma de Números

"""Soma de números"""

soma = 0

while True:
    numero = float(input("Digite um número (0 para parar): "))
    
    if numero == 0:
        break
    
    soma += numero

print("Soma total:", soma)

Número Primo

"""Número primo"""

numero = int(input("Digite um número: "))

if numero <= 1:
    print("Não é primo")
else:
    primo = True
    for i in range(2, numero):
        if numero % i == 0:
            primo = False
            break

    if primo:
        print("É primo")
    else:
        print("Não é primo")

Fatorial

"""Fatorial"""

numero = int(input("Digite um número: "))

fatorial = 1

if numero < 0:
    print("Não existe fatorial de número negativo")
else:
    for i in range(1, numero + 1):
        fatorial *= i
    
    print("Fatorial:", fatorial)

Maior e Menor Valor

"""Maior e menor valor"""

valores = []

for i in range(10):
    num = float(input(f"Digite o {i+1}º valor: "))
    valores.append(num)

maior = max(valores)
menor = min(valores)

print("Maior valor:", maior)
print("Menor valor:", menor)

Lista Ordenada (sem sort)

"""Desafio"""

valores = []

for i in range(10):
    num = float(input(f"Digite o {i+1}º valor: "))

    if len(valores) == 0:
        valores.append(num)
    else:
        inserido = False

        for j in range(len(valores)):
            if num < valores[j]:
                valores.insert(j, num)
                inserido = True
                break

        if not inserido:
            valores.append(num)

print("Lista ordenada:", valores)
